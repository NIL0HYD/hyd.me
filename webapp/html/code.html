// 登录验证
func LoginAuthHandler(rw http.ResponseWriter, req *http.Request) {

	//vars := mux.Vars(req)
	//name := vars["name"]

	email := req.FormValue("email")
	password := req.FormValue("password")

	fmt.Print(password)

	// 根据用户名获取用户基本信息
	result := service.GetUser(email)
	session, _ := store.Get(req, "webapp")
	session.Values["profile"] = result
	session.Save(req, rw)

	tpl := template.Must(template.ParseFiles("html/index.html"))
	tpl.Execute(rw, session.Values["profile"])
}
